The not equals to operator

Description

The not equals to operator

Usage

x %ne% y

Arguments

x

Object.
y

Another object.

Value

A Boolean value.

Examples

AGE <- new_class(class_name = "AGE")
age <- AGE$instantiate
register_method(AGE, ..init.. = function(current) self$current = current)
register_method(AGE, ..ne.. = function(y) self$current != y$current)

patrick <- age(33)
james <- age(33)
patrick %ne% james
